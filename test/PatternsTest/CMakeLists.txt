cmake_minimum_required(VERSION 3.2)

# Enable testing module
enable_testing()

# Find external dependencides
set(GTEST_PACKAGE_NAME gtest)
find_package(${GTEST_PACKAGE_NAME})
if (NOT TARGET ${GTEST_PACKAGE_NAME}::${GTEST_PACKAGE_NAME})
    set(GTEST_PACKAGE_NAME GTest)
    find_package(${GTEST_PACKAGE_NAME})
endif()

# Add project folder into includes
set(CMAKE_INCLUDE_CURRENT_DIR ON)

# Configure test project
set(PATTERNS_TEST_PROJECT PatternsTest)
file(GLOB_RECURSE PATTERNS_TEST_PROJECT_SRC "*.cpp")
file(GLOB_RECURSE PATTERNS_TEST_PROJECT_HDR "*.h")
add_executable(${PATTERNS_TEST_PROJECT} ${PATTERNS_TEST_PROJECT_SRC} ${PATTERNS_TEST_PROJECT_HDR})
target_link_libraries(${PATTERNS_TEST_PROJECT} Patterns 
					  ${GTEST_PACKAGE_NAME}::${GTEST_PACKAGE_NAME})

#Configure source groups
foreach(FILE ${PATTERNS_TEST_PROJECT_SRC} ${PATTERNS_TEST_PROJECT_HDR}) 
    get_filename_component(PARENT_DIR "${FILE}" DIRECTORY)
    string(REPLACE "${CMAKE_CURRENT_SOURCE_DIR}" "" GROUP "${PARENT_DIR}")
    string(REPLACE "/" "\\" GROUP "${GROUP}")

    if ("${FILE}" MATCHES ".*\\.cpp")
       set(GROUP "Source Files${GROUP}")
    elseif("${FILE}" MATCHES ".*\\.h")
       set(GROUP "Header Files${GROUP}")
    endif()

    source_group("${GROUP}" FILES "${FILE}")
endforeach()

# Register tests
add_test(NAME PatternsTest COMMAND ${PATTERNS_TEST_PROJECT})
